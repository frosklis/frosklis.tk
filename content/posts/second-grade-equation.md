$$
\begin{align*}
  & ax^2 + bx + c = 0 \\
  & x^2 + \frac{b}{a}x + \frac{c}{a} = 0 \\
  & \left(x + \frac{b}{2a}\right)^2 + \frac{c}{a} = \left(\frac{b}{2a}\right)^2 \\
  & \left(x + \frac{b}{2a}\right)^2 = \frac{b^2-4ac}{4a^2} \\
  & x \pm \frac{b}{2a} = \sqrt{\frac{b^2-4ac}{4a^2}} \\
  & x =  \frac{-b \pm \sqrt{b^2-4ac}}{2a} \\
\end{align*}
$$
